enum turokDifficulty_e {
    td_easy = 0,
    td_medium,
    td_hard,
    td_hardcore
}

final class TurokGame : KexGame {
    bool        bNewGame;
    int         difficulty;
    MenuManager menu;

    //
    // TurokGame::TurokGame
    //
    
    TurokGame(void) {
        bNewGame = true;
        difficulty = td_medium;
    }
    
    //
    // TurokGame::~TurokGame
    //
    
    ~TurokGame() {
    }
    
    //
    // TurokGame::OnSpawn
    //
    
    void OnSpawn(void) {
        playerHud.Init();
    }
    
    //
    // TurokGame::OnInput
    //
    
    bool OnInput(int type, int data1, int data2, int data3) {
        if(menu.Responder(type, data1, data2, data3)) {
            return true;
        }
        return false;
    }
    
    //
    // TurokGame::OnTick
    //
    
    void OnTick(void) {
    }
    
    //
    // TurokGame::OnLocalTick
    //
    
    void OnLocalTick(void) {
    }
    
    //
    // TurokGame::OnShutdown
    //
    
    void OnShutdown(void) {
    }
};
